def add(a, b):
    """
    Adds two numbers and returns the sum.
    
    Parameters:
    a (int or float): The first number to add.
    b (int or float): The second number to add.
    
    Returns:
    int or float: The sum of the two input numbers.
    
    Examples:
    >>> add(2, 3)
    5
    >>> add(4.5, 2.7)
    7.2
    """
    return a + b
